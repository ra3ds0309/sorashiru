function drawTyphoonOnMap(dataArray) { if (!dataArray || dataArray.length < 2) return; const latLngs = []; const pointsForTube = []; // チューブ計算用のデータストック dataArray.forEach((block) => { if (block.part === "title" || !block.position || !block.position.deg) return; const lat = block.position.deg[0]; const lng = block.position.deg[1]; latLngs.push([lat, lng]); const isAnalysis = (block.part === "実況" || (block.part && block.part.jp === "実況")); let radiusKm = 0; let warningKm = 0; if (!isAnalysis) { // 【予報円】の半径 if (block.probabilityCircleRadius && block.probabilityCircleRadius.km) { radiusKm = block.probabilityCircleRadius.km; } // 【暴風警戒域】の半径（気象庁のJSONから取得） if (block.stormWarningArea && block.stormWarningArea.radius && block.stormWarningArea.radius.km) { warningKm = block.stormWarningArea.radius.km; } else if (block.stormWarningArea && block.stormWarningArea.km) { warningKm = block.stormWarningArea.km; } } else { // 【実況】の場合は、現在の暴風域を暴風警戒域チューブのスタート地点にする if (block.galeWarning && block.galeWarning[0] && block.galeWarning[0].range) { warningKm = block.galeWarning[0].range.km; } } pointsForTube.push({ lat, lng, radiusKm, warningKm, isAnalysis, block }); }); // 🗺️ --- 1. 面と線（チューブ）の描画 --- if (pointsForTube.length > 1) { const leftSides = []; const rightSides = []; const warningLeft = []; const warningRight = []; // セグメント（区間）ごとに計算し、急カーブでの線の交差（崩れ）を防ぐ for (let i = 0; i < pointsForTube.length - 1; i++) { const p1 = pointsForTube[i]; const p2 = pointsForTube[i + 1]; // 2点間の進行方向角度 const angle = Math.atan2( p2.lng - p1.lng, (p2.lat - p1.lat) * Math.cos(((p1.lat + p2.lat) / 2) * (Math.PI / 180)) ); // 指定した半径分、左右に座標をズラすヘルパー関数 const getOffset = (p, rKm, ang) => { if (rKm <= 0) return [p.lat, p.lng]; const meters = rKm * 1000; const R_e = 6378137; const r_lat = (meters / R_e) * (180 / Math.PI); const r_lng = (meters / (R_e * Math.cos(p.lat * (Math.PI / 180)))) * (180 / Math.PI); return [p.lat + r_lat * Math.cos(ang), p.lng + r_lng * Math.sin(ang)]; }; // 予報円チューブ（白）の左右座標をストック leftSides.push(getOffset(p1, p1.radiusKm, angle - Math.PI / 2), getOffset(p2, p2.radiusKm, angle - Math.PI / 2)); rightSides.push(getOffset(p1, p1.radiusKm, angle + Math.PI / 2), getOffset(p2, p2.radiusKm, angle + Math.PI / 2)); // 暴風警戒域チューブ（赤）の左右座標をストック if (p1.warningKm > 0 || p2.warningKm > 0) { warningLeft.push(getOffset(p1, p1.warningKm, angle - Math.PI / 2), getOffset(p2, p2.warningKm, angle - Math.PI / 2)); warningRight.push(getOffset(p1, p1.warningKm, angle + Math.PI / 2), getOffset(p2, p2.warningKm, angle + Math.PI / 2)); } } // 先に暴風警戒域（赤）を描画（予報円の下に敷くため） if (warningLeft.length > 0) { const warningAreaCoords = [...warningLeft, ...[...warningRight].reverse()]; L.polygon(warningAreaCoords, { color: 'transparent', fillColor: '#ff3b30', fillOpacity: 0.15 }).addTo(map); L.polyline(warningLeft, { color: '#ff3b30', weight: 2, opacity: 0.8 }).addTo(map); L.polyline(warningRight, { color: '#ff3b30', weight: 2, opacity: 0.8 }).addTo(map); } // 次に予報円（白）を描画 const areaCoords = [...leftSides, ...[...rightSides].reverse()]; L.polygon(areaCoords, { color: 'transparent', fillColor: '#ffffff', fillOpacity: 0.04 }).addTo(map); L.polyline(leftSides, { color: '#ffffff', weight: 2, opacity: 0.6 }).addTo(map); L.polyline(rightSides, { color: '#ffffff', weight: 2, opacity: 0.6 }).addTo(map); } // 📍 --- 2. 各ポイントの円マークを描画 --- pointsForTube.forEach((p) => { if (p.isAnalysis) { // 【現在地マーク】 L.circleMarker([p.lat, p.lng], { radius: 7, color: '#ff3b30', fillColor: '#ff3b30', fillOpacity: 1, weight: 2 }).addTo(map); // 【実況：暴風域（オレンジ）】元のコードのまま if (p.block.galeWarning && p.block.galeWarning[0] && p.block.galeWarning[0].range) { L.circle([p.lat, p.lng], { radius: p.block.galeWarning[0].range.km * 1000, color: '#ff9500', weight: 1.5, fillColor: '#ff9500', fillOpacity: 0.12 }).addTo(map); } } else { // 【予報：暴風警戒域の円（赤）】キャップを丸くするため枠線なしで塗りつぶし if (p.warningKm > 0) { L.circle([p.lat, p.lng], { radius: p.warningKm * 1000, color: 'transparent', weight: 0, fillColor: '#ff3b30', fillOpacity: 0.15 }).addTo(map); } // 【予報：予報円（白）】 if (p.radiusKm > 0) { L.circle([p.lat, p.lng], { radius: p.radiusKm * 1000, color: '#ffffff', weight: 1.5, dashArray: '5, 5', fillColor: '#ffffff', fillOpacity: 0.03 }).addTo(map); L.circleMarker([p.lat, p.lng], { radius: 2.5, color: '#ffffff', fillColor: '#ffffff', fillOpacity: 1 }).addTo(map); } } }); // 🛤️ --- 3. 中心線の描画 --- if (latLngs.length > 1) { L.polyline(latLngs, { color: '#ffffff', weight: 3, opacity: 0.7 }).addTo(map); } // 📐 --- 4. 描画範囲の自動フィット --- if (latLngs.length > 0) { map.fitBounds(L.latLngBounds(latLngs), { padding: [260, 260] }); } }
